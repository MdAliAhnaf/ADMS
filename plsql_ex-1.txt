2.
declare
 dno dept.deptno%type:=:Dept_No;
 t_emp number(10);
 sum_sal emp.sal%type;
begin
 select count(*), sum(sal) into t_emp, sum_sal from emp where deptno=dno;
 dbms_output.put_line('Total emp = '||t_emp);
 dbms_output.put_line('Sum of salary = '||sum_sal);
end;
